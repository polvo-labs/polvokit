# FormSpacer

A VerticalSpacer for forms.

## Usage

```js
import { Form } from 'react-final-form'
import { FormSpacer, FormField, ButtonSubmit } from 'polvokit'

function FormLogin (props) {
  return (
    <Form {...props}>
      {({ handleSubmit, submitting }) => (
        <form onSubmit={handleSubmit}>
          <FormSpacer>
            <FormField.Input
              name='name'
            />          
            <FormField.Input
              name='password'
            />
            <ButtonSubmit
              label='Login'
              submitting={submitting}
            />
          </FormSpacer>
        </form>
      )}
    </Form>
  )
}
```
